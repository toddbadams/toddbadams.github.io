The Algorithm

Length = RoundUp(log 2 ( decimal ) )
Bits = new Array[Length]
For (i = 0 to Length-1)
If  ( decimal >= 2^ (Length-i) )
	Bits[i]  =  1
	Decimal = decimal - 2^ (Length-i)
Else
	Bits[i] = 0
